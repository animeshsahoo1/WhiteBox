def strategy(indicators):
    """
    SMA Crossover with Trailing Stop (Conservative)
    
    Entry: Fast SMA above slow SMA
    Exit: Fast SMA below slow SMA OR trailing stop hit
    Trailing Stop: 8% from highest price since entry (wider for this volatility)
    Position Size: 50% of capital (conservative)
    """
    import numpy as np
    
    sma_10 = indicators.get('sma_10')
    sma_50 = indicators.get('sma_50')
    close = indicators.get('close')
    
    if any(v is None or np.isnan(v) for v in [sma_10, sma_50, close]):
        return None
    
    # Death cross - exit
    if sma_10 < sma_50:
        return {'action': 'SELL'}
    
    # Golden cross - enter long with wider trailing stop
    if sma_10 > sma_50:
        return {
            'action': 'BUY',
            'trailing_stop': 0.08,  # 8% trailing stop (wider)
            'size': 0.5              # 50% capital
        }
    
    return None
