def strategy(indicators):
    """
    RSI and MACD Strategy
    
    Entry: RSI < 28 and MACD bullish
    Exit: RSI > 72
    SL: 1.5%
    TP: 4%
    """
    import numpy as np
    
    rsi = indicators.get('rsi_14')
    macd = indicators.get('macd')
    macd_signal = indicators.get('macd_signal')
    close = indicators.get('close')
    
    if rsi is None or np.isnan(rsi) or close is None or macd is None or macd_signal is None:
        return None
    
    if rsi < 28 and macd > macd_signal:
        return {
            'action': 'BUY',
            'stop_loss': close * 0.985,
            'take_profit': close * 1.04,
            'size': 0.5
        }
    
    if rsi > 72:
        return {'action': 'SELL'}
    
    return None
