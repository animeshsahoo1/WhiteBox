def strategy(indicators):
    """
    Bollinger Band Breakout with Risk Management
    
    Entry: Price breaks above upper band (momentum)
    Exit: Price drops below middle band OR SL/TP hit
    SL: At lower band
    TP: 2x the band width above entry
    Position Size: 30% of capital (conservative)
    """
    import numpy as np
    
    close = indicators.get('close')
    bb_upper = indicators.get('bb_upper')
    bb_middle = indicators.get('bb_middle')
    bb_lower = indicators.get('bb_lower')
    
    if any(v is None or np.isnan(v) for v in [close, bb_upper, bb_middle, bb_lower]):
        return None
    
    band_width = bb_upper - bb_lower
    
    # Breakout above upper band - momentum entry
    if close > bb_upper:
        return {
            'action': 'BUY',
            'stop_loss': bb_lower,                    # Stop at lower band
            'take_profit': close + (2 * band_width), # TP = 2x band width
            'size': 0.3                               # Conservative 30%
        }
    
    # Price below middle band - exit
    if close < bb_middle:
        return {'action': 'SELL'}
    
    return None
