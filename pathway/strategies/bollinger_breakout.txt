def strategy(indicators):
    """
    Bollinger Bands Breakout Strategy
    
    Momentum strategy that catches volatility expansions.
    Enters when price breaks above upper band.
    
    Entry: Price > Upper Bollinger Band (momentum breakout)
    Exit: Price < Middle Band OR Stop-Loss/Take-Profit hit
    
    Risk Management:
    - Stop Loss: Lower Bollinger Band (dynamic)
    - Take Profit: 2x band width above entry
    - Position Size: 30% (conservative for breakouts)
    """
    import numpy as np
    
    close = indicators.get('close')
    bb_upper = indicators.get('bb_upper')
    bb_middle = indicators.get('bb_middle')
    bb_lower = indicators.get('bb_lower')
    
    if any(v is None for v in [close, bb_upper, bb_middle, bb_lower]):
        return None
    if any(np.isnan(v) for v in [close, bb_upper, bb_middle, bb_lower]):
        return None
    
    band_width = bb_upper - bb_lower
    
    # Entry: Breakout above upper band
    if close > bb_upper:
        return {
            'action': 'BUY',
            'stop_loss': bb_lower,
            'take_profit': close + (2 * band_width),
            'size': 0.3
        }
    
    # Exit: Price falls below middle band
    if close < bb_middle:
        return {'action': 'SELL'}
    
    return None
