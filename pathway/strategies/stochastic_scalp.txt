def strategy(indicators):
    """
    Stochastic Oversold Bounce (Short-term Scalping)
    
    Entry: Stochastic %K < 20 AND %K crosses above %D
    Exit: Stochastic %K > 80 OR quick profit target
    SL: Tight 1% (scalping)
    TP: 2% quick profit
    Position Size: 100% (short holding period)
    """
    import numpy as np
    
    stoch_k = indicators.get('stoch_k')
    stoch_d = indicators.get('stoch_d')
    close = indicators.get('close')
    
    if any(v is None or np.isnan(v) for v in [stoch_k, stoch_d, close]):
        return None
    
    oversold = stoch_k < 20
    overbought = stoch_k > 80
    bullish_cross = stoch_k > stoch_d
    
    # Enter on oversold bounce
    if oversold and bullish_cross:
        return {
            'action': 'BUY',
            'stop_loss': close * 0.99,     # Tight 1% stop
            'take_profit': close * 1.02,   # Quick 2% target
            'size': 1.0
        }
    
    # Exit on overbought
    if overbought:
        return {'action': 'SELL'}
    
    return None
