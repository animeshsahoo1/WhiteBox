def strategy(indicators):
    """
    Simple SMA Crossover (Backward Compatible)
    
    Classic strategy format - just returns BUY/SELL strings.
    No SL/TP (relies on signal reversal to exit).
    """
    import numpy as np
    
    sma_10 = indicators.get('sma_10')
    sma_50 = indicators.get('sma_50')
    
    if sma_10 is None or sma_50 is None:
        return None
    if np.isnan(sma_10) or np.isnan(sma_50):
        return None
    
    if sma_10 > sma_50:
        return 'BUY'
    elif sma_10 < sma_50:
        return 'SELL'
    
    return None
