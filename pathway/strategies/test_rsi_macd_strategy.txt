def strategy(indicators):
    """
    RSI & MACD Strategy
    
    Entry: RSI < 25 and MACD bullish
    Exit: RSI > 75
    SL: 2%
    """
    import numpy as np
    
    rsi = indicators.get('rsi_14')
    macd = indicators.get('macd')
    macd_signal = indicators.get('macd_signal')
    close = indicators.get('close')
    
    if rsi is None or np.isnan(rsi) or close is None or macd is None or macd_signal is None:
        return None
    
    # Entry signal
    if rsi < 25 and macd > macd_signal:
        return {
            'action': 'BUY',
            'stop_loss': close * 0.98,
            'size': 0.5
        }
    
    # Exit signal
    if rsi > 75:
        return {'action': 'SELL'}
    
    return None
