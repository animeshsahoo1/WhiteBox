# interval: 1h
# lookback: 1mo

def strategy(indicators):
    """
    RSI Mean Reversion Strategy
    
    A contrarian strategy that buys oversold conditions.
    Works well in ranging/sideways markets.
    
    Entry: RSI < 30 (oversold - price likely to bounce)
    Exit: RSI > 70 (overbought) OR Stop-Loss/Take-Profit hit
    
    Risk Management:
    - Stop Loss: 2% below entry
    - Take Profit: 5% above entry (2.5:1 reward/risk)
    - Position Size: 50% of capital
    
    Best for: Hourly charts, ranging markets
    """
    import numpy as np
    
    rsi = indicators.get('rsi_14')
    close = indicators.get('close')
    
    if rsi is None or close is None:
        return None
    if np.isnan(rsi) or np.isnan(close):
        return None
    
    # Entry: RSI below 30 = oversold
    if rsi < 30:
        return {
            'action': 'BUY',
            'stop_loss': 0.02,
            'take_profit': 0.05,
            'size': 0.5
        }
    
    # Exit: RSI above 70 = overbought
    if rsi > 70:
        return {'action': 'SELL'}
    
    return None
